## Dick Traits

seed_dick_trait_depending_on_father_and_mother_effect = {
    if = {
        limit = {
            exists = real_father
            exists = mother
        }
        seed_parent_bno_dick_size_effect = {
            PARENT = real_father
        }
        seed_parent_bno_dick_size_effect = {
            PARENT = mother
        }
        set_variable = {
            name = bno_dick_size
            value = bno_average_dick_size_level
        }
    }
    else = {
        seed_bno_dick_size_effect = yes
    }
}

seed_parent_bno_dick_size_effect = {
    if = {
        limit = {
            $PARENT$ = {
                NOT = { has_variable = bno_dick_size }
            }
        }
        $PARENT$ = {
            seed_bno_dick_size_effect = yes
        }
    }
}

seed_dick_trait_effect = {
    if = {
        limit = {
            carn_should_have_dick_trigger = yes
        }
        remove_dick_trait_effect = yes
        switch = {
            trigger = var:bno_dick_size
            1 = { add_trait = dick_small_bad_3 }
            2 = { add_trait = dick_small_bad_2 }
            3 = { add_trait = dick_small_bad_1 }
            5 = { add_trait = dick_big_good_1 }
            6 = { add_trait = dick_big_good_2 }
            7 = { add_trait = dick_big_good_3 }
        }
    }
}

remove_dick_trait_effect = {
    remove_trait = dick_small_bad_3
    remove_trait = dick_small_bad_2
    remove_trait = dick_small_bad_1
    remove_trait = dick_big_good_1
    remove_trait = dick_big_good_2
    remove_trait = dick_big_good_3
}

seed_bbc_trait_effect = {
    if = {
        limit = {
            has_trait = bno_blood_4
            has_trait = dick_big_good_3
            NOT = { has_trait = bno_bbc }
        }
        add_trait = bno_bbc
    }
    else_if = {
        limit = {
            has_trait = bno_blood_3
            has_trait = dick_big_good_3
            NOT = { has_trait = bno_bbc }
        }
        random = {
            chance = 75
            add_trait = bno_bbc
        }
    }
    else_if = {
        limit = {
            has_trait = bno_blood_2
            has_trait = dick_big_good_3
            NOT = { has_trait = bno_bbc }
        }
        random = {
            chance = 50
            add_trait = bno_bbc
        }
    }
    else_if = {
        limit = {
            has_trait = bno_blood_1
            has_trait = dick_big_good_3
            NOT = { has_trait = bno_bbc }
        }
        random = {
            chance = 25
            add_trait = bno_bbc
        }
    }
}

make_bbc_inactive_for_non_futa_effect = {
    if = {
        limit = {
            has_trait = bno_bbc
            carn_should_have_dick_trigger = no
        }
        make_trait_inactive = bno_bbc
    }
}

bno_seed_dick_traits_effect = {
    seed_dick_trait_depending_on_father_and_mother_effect = yes
    seed_dick_trait_effect = yes
    seed_bbc_trait_effect = yes
    make_bbc_inactive_for_non_futa_effect = yes
}

## African Traits

seed_african_trait_effect = {
    if = {
        limit = {
            is_culture_african_trigger = yes
            age > 0
            NOT = { has_trait = bno_blood }
        }
        add_trait = bno_blood_4
    }
}

seed_african_blood_concentration_effect = {
    if = {
        limit = {
            NOT = { has_variable = bno_blood_concentration }
        }
        switch = {
            trigger = has_trait
            bno_blood_1 = {
                set_variable = {
                    name = bno_blood_concentration
                    value = 1
                }
            }
            bno_blood_2 = {
                set_variable = {
                    name = bno_blood_concentration
                    value = 2
                }
            }
            bno_blood_3 = {
                set_variable = {
                    name = bno_blood_concentration
                    value = 3
                }
            }
            bno_blood_4 = {
                set_variable = {
                    name = bno_blood_concentration
                    value = 4
                }
            }
            fallback = {
                set_variable = {
                    name = bno_blood_concentration
                    value = 0
                }
            }
        }
    }
}

give_average_blood_concentration_to_child = {
    set_variable = {
        name = bno_blood_concentration
        value = bno_average_blood_level
    }
}
